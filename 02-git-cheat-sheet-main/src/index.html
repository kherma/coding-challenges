git cheat sheet

Basic Operations

git init

git add . // Delete file from staging area git restore —staged name

git status git diff

git commit -m ‘message’ // Change author of last commit git commit —amend -author ’...’ // Change message of last commit git commit —amend -m ‘...’ // Delete from working repo and staging area git rm 'file_name'

// git fetch + git merge FETCH_HEAD git pull -v git push -u git fetch -v

Branch Management

git branch -a git branch 'name' git checkout 'name' git checkout -b 'name'

git branch -d 'name' git branch -m 'old' 'new'

// If branch deleted in remote repo git fetch -v git remote update origin —prune git branch -d 'branch'

// Push local branch to remote repo git push -u origin 'branch'

// Delete remote branch from terminal git push update origin -d 'branch'

Reading

// Status of remote / local repos git remote show origin git remote -vv

git cat-file -s / -t / -p

git log —oneline / —graph

// All changes from last 90 days git reflog

Tags

git tag

git tag v1.0.0 git tag -a v1.0.0 -m ‘message’

git push -u —tag git push -u origin v1.0.0

Merge & Rebase

git merge 'name'

git rebase 'name' | git merge 'name'

Stash

git stash

git stash pop git stash drop

Reset & Revert

git reset —hard HASH git reset -mixed HASH git reset -soft HASH

git revert HASH

Cherry pick

git cherry-pick hash

git cherry pick —no-commit HASH

Packages, ignore & config

git gc git 'file' | git unpack-objects

.gitignore git rm —cached 'file'

git config —list git config — global user.name ‘name’ git config credential.helper “”

Fork

// Fork on GitHub // Clone Repo // Pull request

git remote add upstream url git fetch upstream -v git pull upstream main git push -u
